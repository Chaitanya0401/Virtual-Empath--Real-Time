def extract_mfcc(wav_file_name):

    y, sr = librosa.load(wav_file_name)
    mfccs = np.mean(librosa.feature.mfcc(y=y, sr=sr, n_mfcc=40).T,axis=0)
    
    return mfccs

model.add(Dense(units=32, activation='relu'))
model.add(Dropout(0.3))
